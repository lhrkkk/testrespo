Store Management

(defonce ref-store
  (atom
    {:router nil
     :tasks []
     :states {}}))

(defn updater [store op op-data]
  (case op
    :states (update store :states (mutate op-data))
    :add-task (add-task store op-data)
    store))

(defn dispatch! [op op-data]
  (let [new-store (updater @ref-store op op-data)]
    (reset! ref-store new-store)))

(add-watch ref-store
  (fn [] (println "store is changed!")))
State Management

(defn on-input [state cursor]
  (fn [e dispatch!]
    ; dispatch :states action with cursor and new state
    (dispatch! :states [cursor (:value e)])))

(def comp-field
  (create-comp :field
    (fn [states]
      (fn [cursor]
        (let
          ; states are inside store as a states tree
          ; use `or` to specify default state
          [state (or (:data states) "")]
          (div {}
            (div {}
              (comp-text "name" nil))
            (input {:event {:input (on-input states cursor)}
                    :attrs {:placeholder "name"
                            :value state}})))))))
